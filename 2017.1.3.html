实例一:切换选项改变显示图片,主要代码如下: 

	    $(function(){
		$(".tab>li").hover(function(){
			$(this).addClass("active").siblings("li").removeClass("active");  //移入改变选项卡的样式
			var ind=$(this).index();  //获取选中选项卡的索引号
			$(".products div").eq(ind).addClass("selected").siblings("div").removeClass("selected"); //根据索引号改变图片的显示
		})
	})

实例二:切换选项使图片飞入,主要代码如下: 

	    $(function(){
		var n=1;
		$("ol>li").mouseenter(function(){
			n++;
			$(this).addClass("active").siblings("li").removeClass("active");
			var ind=$(this).index();
			$("ul li").eq(ind).stop().css({
				"z-index":n,   //改变需要飞入的图片的层级
				"left":"300px"
			}).animate({
				
				"left":"0px"
			},500)
		})
	})

效果如下图所示

实例三:移入改变图片显示大小,主要代码如下: 

	     $(function () {
            $(".wrap li").each(function (i,ele) {  //通过each方法添加背景图片
                $(this)
                        .css("background-image","url(img/" + (i + 1) +".jpg)");
            }).mouseenter(function () {
                $(this).stop().animate({
                    "width":600
                },300);
                $(this).siblings().stop().animate({
                    "width":100
                },300)
            }).mouseleave(function () {
                $(".wrap li").stop().animate({
                    "width":200
                },300)
            });
        })
