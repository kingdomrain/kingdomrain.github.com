canvas

也就是画布HTML5 < canvas> 标签用于绘制图像（通过脚本，通常是 JavaScript）。 不过，< canvas> 元素本身并没有绘制能力（它仅仅是图形的容器） - 您必须使用脚本来完成实际的绘图任务。 getContext() 方法可返回一个对象，该对象提供了用于在画布上绘图的方法和属性。

比较常用的几种canvas方法
var canvas = document.getElementById("canvas");     //西瓜,圆的写法
var can = canvas.getContext("2d")
can.rect(0,0,110,110);
can.lineWidth="10";
can.strokeStyle="red";
can.stroke();
can.fillStyle="#00ff00";
can.fillRect(5,5,100,100);
can.beginPath();
can.arc(200, 200, 100, 0.5 * Math.PI, 1.5 * Math.PI);
can.lineWidth = "5";
can.strokeStyle = "green";
can.closePath();
can.stroke();
can.fillStyle = "red";
can.fill();
for(var i = 0; i < 100; i++) {   
var n = Math.random() * 100;
var l = Math.random() * 200;
var x = l - 100;
if(n * n + x * x < 100 * 100) {
can.beginPath();
can.arc(200 - n, 300 - l, 2, 0.5 * Math.PI, 1.5 * Math.PI);
can.fillStyle = "black";
can.fill();
}
}
