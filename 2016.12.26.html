常用的事件

	
	$(document).ready()  //相当于页面载入时
	$(document).click()  //点击事件
	$(document).dbclick()  //双击事件
	$(document).change()  //selected选项改变事件
	$(document).focus()  //获取焦点事件
	$keydown([[data],fn])   //键盘按下事件
	$keypress([[data],fn])  //键盘点击事件
	$keyup([[data],fn])   //键盘松开事件
	$mousedown([[data],fn])   //鼠标按下事件
	$mouseenter([[data],fn])  //鼠标移入事件
	$mouseleave([[data],fn])  //鼠标移开事件,与 mouseout 事件不同，只有在鼠标指针离开被选元素时，才会触发 mouseleave 事件。如果鼠标指针离开任何子元素，同样会触发 mouseout 事件。
	$mousemove([[data],fn])  //鼠标移动事件
	$mouseout([[data],fn])  //鼠标移出事件
	$mouseover([[data],fn]) //鼠标经过事件
	$mouseup([[data],fn])  //鼠标松开事件
	$resize([[data],fn])  //窗口大小改变事件
	$scroll([[data],fn])  //滚动条事件
	$select([[data],fn]) //当 textarea 或文本类型的 input 元素中的文本被选择事件
	$submit([[data],fn]) //提交表单事件
	$unload([[data],fn]) //用户离开页面事件
