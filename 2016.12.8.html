string对象

string的申明方式

			var str = "asd ";
			var str = " ";
			var str1=new String();
			var str2=new String("asd")

比较常用的几种string对象方法

			match() 找到一个或多个正则表达式的匹配。 如:
			var str="asd";
			console.log(str.match("s")) //输出结果为s
			
			indexOf() 检索字符串 .如:
			var str="asd";
			console.log(str.indexOf("s")) //输出结果为1
			
			substr() 从起始索引号提取字符串中指定数目的字符。 如:
			var str="asd";
			console.log(str.substr(1,2)) //输出结果为sd
			
			substring() 提取字符串中两个指定的索引号之间的字符。 如:
			var str="asd";
			console.log(str.substring(1,2)) //输出结果为s
			该方法可以写负数
			
			slice() 提取字符串的片断，并在新的字符串中返回被提取的部分。 如:
			var str="asd";
			console.log(str.slice(1,2)) //输出结果为s
			该方法也可写负数
			
			split() 把字符串分割为字符串数组。 如:
			var str="asd";
			console.log(str.split("s")) //输出结果为a,d
			
			charAt() 返回在指定位置的字符。  如:
			var str="asd";
			console.log(str.charAt(1))//输出结果为s
			
			replace() 替换与正则表达式匹配的子串。 如:
			stringObject.replace(regexp/substr,replacement)
数组对象

数组的声明方式

			var arr=[1,2];
			var arr1=[];
			var arr2=new Array();
			var arr3=new Array(1,2);
			var arr3=new Array(1); //如果括号中只写一个数字的话代表数组的长度

比较常用的几种string对象方法

			slice() 从某个已有的数组返回选定的元素 如:
			var arr=[1,2,3,4];
            console.log(arr.slice(2))  //输出结果为3,4
            
            splice() 删除元素，并向数组添加新元素。 如:
            var arr=[1,2,3,4];
            arr.splice(2)
            console.log(arr)  //输出结果为1,2,改方法会改变原数组
            
            concat() 连接两个或更多的数组，并返回结果。 如:
            var arr=[1,2,3,4];
			var arr1=[2,3,4,5]
			console.log(arr.concat(arr1)) //输出结果为1,2,3,4,2,3,4,5
			
			join() 把数组的所有元素放入一个字符串。元素通过指定的分隔符进行分隔。 如:
			var arr=[1,2,3,4];
			console.log(arr.join(3)) //输出结果为1323334
			
			pop() 删除并返回数组的最后一个元素 如:
			var arr=[1,2,3,4];
			arr.pop()
			console.log(arr)
			
			push() 向数组的末尾添加一个或更多元素，并返回新的长度。 如:
			var arr=[1,2,3,4];
			console.log(arr.push(5)) //输出结果为5
			
			reverse() 颠倒数组中元素的顺序。 如:
			var arr=[1,2,3,4];
			console.log(arr.reverse()) //输出结果为4,3,2,1
			
			shift() 删除并返回数组的第一个元素  如:
			var arr=[1,2,3,4];
			console.log(arr.shift()+"  "+arr) //输出结果为1  2,3,4
			
			sort() 对数组的元素进行排序 如:
			var arr=[1,5,3,2];
			console.log(arr.sort()) //输出结果为1,2,3,5
			
			unshift() 向数组的开头添加一个或更多元素，并返回新的长度。 如:
			var arr=[1,5,3,4];
			console.log(arr.unshift(7)+"  "+arr) //输出结果为5  7,1,5,3,4
