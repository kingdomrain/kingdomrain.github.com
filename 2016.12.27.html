小实例，通过ajax获取网站上的数据

	
$("img").click(
	function(){
		
			var _data={};
			_data.pageNum=2;
			_data.pageSize=20;
			$.ajax({   
			type:"get",
			url:"http://www.hemijiaoyu.com/service/searchApply",
			async:true,
			data:_data,
			dataType:"JSON",
			success:function(msg){
				//console.log(msg.list[0]);
				var str="< table>";
				for(var i=0;i< msg.list.length;i++){
					str+="";
					
					for(var j=0 in msg.list[i]){
						str+=""+msg.list[i][j]+"";
					}
					str+=""
				}
				str+="";
				$("div").html(str)
			},
			eerror:function(msg){
				console.log("错误")
			}
		});
		
	}
)
